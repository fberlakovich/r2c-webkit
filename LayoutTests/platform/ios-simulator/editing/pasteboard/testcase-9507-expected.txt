"
| <div>
|   class="editing"
|   id="test"
|   "
<#selection-anchor>foo
"
|   <div>
|     style="color: rgb(255, 0, 0);"
|     "
"
|     <div>
|       "bar<#selection-focus>"
|     "
"
|   "
"
| "
"
